COMPRENSIÓN DE ARCHIVOS
#Primero para visualizar los archivos
ls dump4.sql -l
#más detalles
ls dump4.sql -lh
#Para comprimirlo
gzip dump4.sql
#Para buscarlo
ls dump4*
#Para devolverlo a la normalidad
ls -d dump4.sql.gz
#Crear archivo
cf

COMBINACIÓN DE ARCHIVOS
#Crear nuevo archivo donde van a gruparse otros
tar cf backup.tar backup/*
#Ver el contenido
ls backup/ -l
#Ver contenido del archivo
tar tf backup.tar
#Para expandirlo o sacarlo del agrupamiento
tar xf backup.tar
ls -l
#Juntar archivos y comprimir resultados de la combinación
tar czf backup.tgz backup/*
